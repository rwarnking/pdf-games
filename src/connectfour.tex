const GAME_DESC = "Connect four is a connection board game. The goal is to place a set amount\
of stones (4) with the given playercolor in such a way that they lie on a strait line.\
Each turn a player can choose a column in which his stone should be placed.\
The stones will be placed at the bottom of the column or on top of the current top stone.";

% Outside variables
const FIELD_WIDTH = AFMakeNumber(\fieldWidth);
const FIELD_HEIGHT = AFMakeNumber(\fieldHeight);
const PLAYER_COUNT = AFMakeNumber(\playerCount);
const WINNING_PIECES = AFMakeNumber(\winningPieces);

% Constants
const EMPTY = 0;

% Globals
% The game field array
var complField = new Array();
var activePlayer = 0;
var gameover = false;

% Color data
const COLORS = new Array(
    color.black,
    [ "RGB", 0.843, 0.078, 0.058 ],
    [ "RGB", 0.203, 0.603, 0.054 ],
    [ "RGB", 0.078, 0.305, 0.560 ],
    [ "RGB", 0.560, 0.078, 0.533 ]
);

const HIGHLIGHT_COLOR = [ "RGB", 0.984, 0.960, 0.176 ];

%%%%%%%%%%%%%%%%%%
% Initialisation %
%%%%%%%%%%%%%%%%%%
/**
 * Initialise the default game state.
 */
function initialise() {
    for (var row = 0; row < FIELD_HEIGHT; row++) {
        var tmp = new Array();
        for (var column = 0; column < FIELD_WIDTH; column++) {
            tmp.push(EMPTY);
        }
        complField.push(tmp);
    }

    this.getField("info").fillColor = color.transparent;
    this.getField("info").borderColor = color.transparent;

    printInfoText("Player " + (activePlayer + 1) + ": Please select a column.");

    % Init the GUI
    initialiseButtons();

    drawAll();
}

/**
 * Initialise all buttons from the GUI.
 * For example add callbacks and set the correct colors.
 */
function initialiseButtons() {
    % Assign each button the coresponding function and disable the highlights
    this.getField("restart").setAction("MouseUp", "restart();");
    this.getField("restart").highlight="none";

    % Each field needs to be connected to an input function
    var funcName = "addToColumn";
    for (var x = 0; x < FIELD_WIDTH; x++) {
        var callbackName = funcName + "(" + x.toString() + ");";
        this.getField("col" + x).setAction("MouseDown", callbackName);
        this.getField("col" + x).highlight="none";
        this.getField("col" + x).borderColor = color.transparent;
    }

    % Tooltip setup
    this.getField("tooltipbtn").setAction("MouseDown", "tooltip(true);");
    this.getField("tooltipbtn").setAction("MouseUp", "tooltip(false);");
    this.getField("tooltipbtn").highlight="none";
}

/**
 * Restarts the game by clearing the field array.
 */
function restart() {
    % Clear complField and redraw
    var field;
    for (var x = 0; x < FIELD_WIDTH; x++) {
        for (var y = 0; y < FIELD_HEIGHT; y++) {
            complField[y][x] = EMPTY;

            draw(x, y);
        }
    }

    activePlayer = 0;
    printInfoText("Restarted - Player " + (activePlayer + 1) + ": Please select a column.");
    gameover = false;
}

/**
 * Enable or disable the tooltip
 *
 * @param {Boolean} on Enable or disable tooltip
 */
function tooltip(on) {
    if (on) {
        % Move the tooltip field to the center of the screen
        % and adjust the size so the text can be displayed
        var field = this.getField("tooltiptxt");
        field.delay = true;
        field.value = GAME_DESC;

        field.multiline = true;
        field.textSize = 16;
        field.fillColor = color.gray;
        % hight 842, width 596
        % upper-left x, upper-left y, lower-right x and lower-right y
        field.rect = [128, 601, 468, 401];

        field.delay = false;
    } else {
        % Remove the field such that it is not visible
        var field = this.getField("tooltiptxt");
        field.delay = true;
        field.value = "";

        field.multiline = false;
        field.fillColor = color.gray;
        field.rect = [0, 0, 0, 0];

        field.delay = false;
    }
}

%%%%%%%%%%%%%
% Gamelogic %
%%%%%%%%%%%%%
/**
 * Add a stone to the selected column (x).
 * Also selects new player and redraws.
 *
 * @param {Number} x The column to place the stone in.
 */
function addToColumn(x) {
    if (gameover) {
        return;
    }

    var y = getYFromX(x);
    % Column is full
    if (y == -1) {
        printInfoText("Column is full. Player " + (activePlayer + 1) + ": Select a valid column.");
        return;
    }
    complField[y][x] = activePlayer + 1;

    checkGameover(x, y);

    activePlayer++;
    % Reset player count
    if (activePlayer >= PLAYER_COUNT) {
        activePlayer -= PLAYER_COUNT;
    }

    if (!gameover) {
        printInfoText("Player " + (activePlayer + 1) + ": Please select a column.");
    }

    draw(x, y);
}

/**
 * Returns the first empty spot for the given column (x).
 *
 * @param {Number} x The column for which the y should be returned.
 */
function getYFromX(x) {
    for (var y = 0; y < FIELD_HEIGHT; y++) {
        if (complField[y][x] != EMPTY) {
            return y - 1;
        }
    }
    return FIELD_HEIGHT - 1;
}

/**
 * Checks whether the current move concludes the game and someone won.
 * The position of the last move is used to check the neighbourhood for
 * potential lines in all possible directions. Sets gameover if
 * a line was found.
 *
 * @param {Number} x The x coordinate
 * @param {Number} y The y coordinate
 */
function checkGameover(x, y) {
    var count = 0;
    var maxCount = 0;
    % Define directions
    var directions = [
        0, 1,
        1, 1,
        1, 0,
        1, -1
    ]
    % Iterate directions
    for (var dirIdx = 0; dirIdx < directions.length; dirIdx += 2) {
        count = 0;
        maxCount = 0;
        % Count succeeding stones
        for (var wp = -WINNING_PIECES + 1; wp < WINNING_PIECES; wp++) {
            var nx = x + directions[dirIdx] * wp;
            var ny = y + directions[dirIdx + 1] * wp;
            if (nx < 0 || nx >= FIELD_WIDTH ||
                ny < 0 || ny >= FIELD_HEIGHT ||
                complField[ny][nx] != activePlayer + 1
            ) {
                count = 0;
            } else {
                count++;
                maxCount = count;
            }
        }
        % If there are enough pieces found, color the winning pieces and set gameover
        if (maxCount >= WINNING_PIECES) {
            gameover = true;
            colorWinner(x, y, dirIdx);
            printInfoText("The winner is: Player " + (activePlayer + 1));
        }
    }
}

%%%%%%%%%%%%%%%%%%%%%
% Draw and coloring %
%%%%%%%%%%%%%%%%%%%%%
/**
 * Draws the complete field.
 */
function drawAll() {
    for (var x = 0; x < FIELD_WIDTH; x++) {
        for (var y = 0; y < FIELD_HEIGHT; y++) {
            draw(x, y);
        }
    }
}

/**
 * Draw the field at the coordinates x, y.
 *
 * @param {Number} x The x coordinate
 * @param {Number} y The y coordinate
 */
function draw(x, y) {
    field = this.getField("pos" + x + "-" + y);
    field.delay = true;
    var value = complField[y][x];
    if (value > EMPTY) {
        field.fillColor = COLORS[value - 1];
    } else {
        field.fillColor = color.white;
        field.borderColor = color.transparent;
    }
    field.delay = false;
}

/**
 * Marks all stones in a line with x, y and the given direction (dirIdx),
 * given the stone has the same player index as the stone at x, y.
 *
 * @param {Number} x The x coordinate
 * @param {Number} y The y coordinate
 * @param {Number} dirIdx The direction index
 */
function colorWinner(x, y, dirIdx) {
    var directions = [
        0, 1,
        1, 1,
        1, 0,
        1, -1
    ]
    % Allows for highlighting all stones left and right from the x, y
    for (var mul = -1; mul < 2; mul += 2) {
        for (var wp = 0; wp < WINNING_PIECES; wp++) {
            var nx = x + directions[dirIdx] * wp * mul;
            var ny = y + directions[dirIdx + 1] * wp * mul;
            if (nx < 0 || nx >= FIELD_WIDTH ||
                ny < 0 || ny >= FIELD_HEIGHT ||
                complField[ny][nx] != activePlayer + 1
            ) {
                wp = WINNING_PIECES;
            } else {
                var field = this.getField("pos" + nx + "-" + ny);
                field.delay = true;
                field.borderColor = HIGHLIGHT_COLOR;
                field.delay = false;
            }
        }
    }
}

/**
 * Print information about the game.
 *
 * @param {String} text Info string
 */
function printInfoText(text) {
    var field = this.getField("info");
    field.delay = true;
    field.value = text;
    field.delay = false;
}
