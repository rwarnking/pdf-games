const GAME_DESC = "Text.\
Text\
Text,\
Text.";

var FIELD_WIDTH = AFMakeNumber(\fieldWidth);
var FIELD_HEIGHT = AFMakeNumber(\fieldHeight);

var complField = new Array();
var gameover = false;

const MINE = -1;
const EMPTY = 0;
const HIDDEN = false;
const VISIBLE = true;

var colors = new Array(
    color.black,
    [ "RGB", 0.843, 0.078, 0.058 ],
    [ "RGB", 0.203, 0.603, 0.054 ],
    [ "RGB", 0.078, 0.305, 0.560 ],
    [ "RGB", 0.560, 0.078, 0.533 ]
);

highlightcolor = [ "RGB", 0.984, 0.960, 0.176 ];

%%%%%%%%%%
% Helper %
%%%%%%%%%%
function getRandomInt(max) {
    return Math.floor(Math.random() * max);
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% Initialisation and Restart %
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
function initialise() {
    for (var row = 0; row < FIELD_HEIGHT; row++) {
        var tmp = new Array();
        for (var column = 0; column < FIELD_WIDTH; column++) {
            tmp.push([HIDDEN, EMPTY]);
        }
        complField.push(tmp);
    }

    initialiseButtons();

    fillMines();
    drawAll();
    printInfoText("Ready to play, good luck.");
}

function initialiseButtons() {
    var funcName = "reveal";
    for (var x = 0; x < FIELD_WIDTH; x++) {
        for (var y = 0; y < FIELD_HEIGHT; y++) {
            var callbackName = funcName + "(" + x.toString() + "," + y.toString() + ");";
            var f = this.getField("pos" + x + "-" + y);
            f.setAction("MouseDown", callbackName);
            f.borderColor = color.black;
            f.fillColor = color.white;
            f.alignment = "center";
        }
    }

    this.getField("restart").setAction("MouseDown", "restart();");
    this.getField("restart").highlight="none";

    this.getField("percentage").currentValueIndices = 2;

    this.getField("info").fillColor = color.transparent;
    this.getField("info").borderColor = color.transparent;

    % Tooltip setup
    this.getField("tooltipbtn").setAction("MouseDown", "tooltip(true);");
    this.getField("tooltipbtn").setAction("MouseUp", "tooltip(false);");
    this.getField("tooltipbtn").highlight="none";
}

/**
 * Enable or disable the tooltip
 *
 * @param {Boolean} on Enable or disable tooltip
 */
function tooltip(on) {
    if (on) {
        % Move the tooltip field to the center of the screen
        % and adjust the size so the text can be displayed
        var field = this.getField("tooltiptxt");
        field.delay = true;
        field.value = GAME_DESC;

        field.multiline = true;
        field.textSize = 16;
        field.fillColor = color.gray;
        % hight 842, width 596
        % upper-left x, upper-left y, lower-right x and lower-right y
        field.rect = [128, 581, 468, 381];

        field.delay = false;
    } else {
        % Remove the field such that it is not visible
        var field = this.getField("tooltiptxt");
        field.delay = true;
        field.value = "";

        field.multiline = false;
        field.fillColor = color.gray;
        field.rect = [0, 0, 0, 0];

        field.delay = false;
    }
}

%%%%%%%%%%%%%%%%%%%%
% Fill and restart %
%%%%%%%%%%%%%%%%%%%%
function fillMines() {
    % Distribute the mines onto the field
    var mines = (FIELD_WIDTH * FIELD_HEIGHT) / 100 * this.getField("percentage").value;
    var x = 0;
    var y = 0;
    for (var i = 0; i < mines; i++) {
        x = getRandomInt(FIELD_WIDTH);
        y = getRandomInt(FIELD_HEIGHT);
        complField[y][x][1] = MINE;
    }

    % For each field that is not a mine calc the mine count
    for (var x = 0; x < FIELD_WIDTH; x++) {
        for (var y = 0; y < FIELD_HEIGHT; y++) {
            if (complField[y][x][1] == MINE) {
                continue;
            }
            for (var xx = x - 1; xx <= x + 1; xx++) {
                for (var yy = y - 1; yy <= y + 1; yy++) {
                    if (xx < 0 || yy < 0 || xx >= FIELD_WIDTH || yy >= FIELD_HEIGHT || (xx == x && yy == y)) {
                        continue;
                    }

                    if (complField[yy][xx][1] == MINE) {
                        complField[y][x][1]++;
                    }
                }
            }
        }
    }
}

function restart() {
    gameover = false;

    % Clear the field
    for (var x = 0; x < FIELD_WIDTH; x++) {
        for (var y = 0; y < FIELD_HEIGHT; y++) {
             complField[y][x] = [HIDDEN, EMPTY];
        }
    }

    fillMines();
    drawAll();
    printInfoText("Restarted the game.");
}

function reveal(x, y) {
    if (gameover) {
        return;
    }

    complField[y][x][0] = VISIBLE;

    if (complField[y][x][1] == MINE) {
        gameover = true;
        printInfoText("You hit a mine and lost!");
    }

    % Reveal all empty spots in the neighbourhood
    if (complField[y][x][1] == EMPTY) {
        var stack = [[x, y]];
        while (stack.length > 0) {
            var elem = stack.pop();
            for (var xx = elem[0] - 1; xx <= elem[0] + 1; xx++) {
                for (var yy = elem[1] - 1; yy <= elem[1] + 1; yy++) {
                    if (xx < 0 || yy < 0 || xx >= FIELD_WIDTH || yy >= FIELD_HEIGHT ||
                        (xx == elem[0] && yy == elem[1]) || complField[yy][xx][0] == VISIBLE) {
                        continue;
                    }
                    % Reveal the field
                    complField[yy][xx][0] = VISIBLE;
                    draw(xx, yy);
                    % If it is a number then dont add it to the stack
                    if (complField[yy][xx][1] != EMPTY) {
                        continue;
                    }
                    stack.push([xx, yy]);

                }
            }
        }
    }

    draw(x, y);
}

%%%%%%%%
% Draw %
%%%%%%%%
function drawAll() {
    for (var x = 0; x < FIELD_WIDTH; x++) {
        for (var y = 0; y < FIELD_HEIGHT; y++) {
            draw(x, y);
        }
    }
}

function draw(x, y) {
    var field = this.getField("pos" + x + "-" + y);
    field.delay = true;

    if (complField[y][x][0] == VISIBLE) {
        if (complField[y][x][1] == MINE) {
            field.fillColor = color.black;
            field.borderColor = color.black;
            field.value = "M";
        } else if (complField[y][x][1] != EMPTY) {
            field.value = complField[y][x][1];
        } else {
            field.fillColor = color.transparent;
            field.borderColor = color.transparent;
        }
    } else {
        field.borderColor = color.black;
        field.fillColor = color.white;
        field.value = "";
    }

    field.delay = false;
}

/**
 * Print information about the winner and the gamestate.
 */
function printInfoText(text) {
    var field = this.getField("info");
    field.delay = true;
    field.value = text;
    field.delay = false;
}
